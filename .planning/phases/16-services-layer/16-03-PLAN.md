  <action>
    Create packages/mcp/src/services/message.service.ts with:

    ```typescript
    import { getPrisma } from '../prisma.js';
    import type { Message } from '../types.js';
    import { randomUUID } from 'node:crypto';

    export class MessageService {
      private prisma = getPrisma();

      // Helper: Convert Prisma Message to domain Message
      private messageToDomain(message: any): Message {
        return {
          id: message.id,
          projectId: message.projectId || undefined,
          role: message.role as Message['role'],
          content: message.content,
          createdAt: message.createdAt.toISOString(),
        };
      }

      /**
       * Add a message to the dialog history
       */
      async add(role: 'user' | 'assistant', content: string, projectId?: string): Promise<Message> {
        const message = await this.prisma.message.create({
          data: {
            id: randomUUID(),
            projectId,
            role,
            content,
          },
        });

        return this.messageToDomain(message);
      }

      /**
       * Get all messages for a project, ordered by creation time
       */
      async list(projectId?: string): Promise<Message[]> {
        const messages = await this.prisma.message.findMany({
          where: projectId ? { projectId } : undefined,
          orderBy: { createdAt: 'asc' },
        });

        return messages.map(m => this.messageToDomain(m));
      }

      /**
       * Delete all messages for a project
       */
      async deleteAll(projectId?: string): Promise<number> {
        const result = await this.prisma.message.deleteMany({
          where: projectId ? { projectId } : {},
        });

        return result.count;
      }
    }

    export const messageService = new MessageService();
    ```

    Key patterns:
    - All database operations use Prisma Client (no getDb())
    - Date conversion: DateTime.toISOString() for createdAt
    - Role enum conversion: Prisma MessageRole → domain 'user' | 'assistant'
    - projectId nullable handling (undefined for global messages)
    - Singleton export matches existing taskStore pattern

    The message.service.ts file will be ~60 lines.
