    ```typescript
      const prisma = (await import('@gantt/mcp/prisma')).getPrisma();
      const project = await prisma.project.findUnique({
        where: { id: req.user!.projectId },
        select: { version: true },

    }
    ```

    **PATCH /api/tasks/:id** — Add schedule change detection:
    ```typescript
    const isScheduleChange = updates.startDate !== undefined || updates.endDate !== undefined;
    if (isScheduleChange) {
      const version = await getProjectVersionForReq(req);
      const cmd = buildCommandFromUpdate(taskId, updates);
      if (cmd) {
        const response = await commandService.commitCommand({
          projectId: req.user!.projectId,
          clientRequestId: randomUUID(),
          baseVersion: version,
          command: cmd,
        }, 'user', req.user!.userId);
        if (response.accepted) {
          return reply.send({
            task: response.result.snapshot.tasks.find(t => t.id === taskId),
            changedTasks: response.result.snapshot.tasks.filter(t => response.result.changedTaskIds.includes(t.id)),
            changedIds: response.result.changedTaskIds,
          });
        }
        return reply.status(409).send(response);
      }
    }
    // Fall through to existing PATCH flow for non-schedule changes
    ```

    Helper `buildCommandFromUpdate(taskId, updates)`:
    - If `startDate` changed and `endDate` unchanged -> `{ type: 'resize_task', taskId, anchor: 'start', date: updates.startDate }`
    - If `endDate` changed and `startDate` unchanged -> `{ type: 'resize_task', taskId, anchor: 'end', date: updates.endDate }`
    - If both changed -> `{ type: 'move_task', taskId, startDate: updates.startDate }`
