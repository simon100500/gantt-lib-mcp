    The new TaskScheduler class must maintain the SAME public interface so TaskService doesn't break:
    ```typescript
    import {
      moveTaskWithCascade,
      resizeTaskWithCascade,
      recalculateTaskFromDependencies,
      recalculateProjectSchedule,
      type ScheduleCommandResult as CoreResult,
      type ScheduleCommandOptions as CoreOptions,
    } from 'gantt-lib/core/scheduling';
    import type { Task, ScheduleCommand, ScheduleCommandOptions, ScheduleCommandResult } from './types.js';

    export class TaskScheduler {
      private snapshot: Task[];
      private options: ScheduleCommandOptions;

      constructor(taskMap: Map<string, Task>, options?: ScheduleCommandOptions) {
        this.snapshot = Array.from(taskMap.values());
        this.options = options ?? {};
      }

      execute(command: ScheduleCommand, options?: ScheduleCommandOptions & { includeSnapshot?: boolean }): ScheduleCommandResult {
        const opts: CoreOptions = {
          businessDays: (options ?? this.options).businessDays,
          weekendPredicate: (options ?? this.options).weekendPredicate,
        };

        let coreResult: CoreResult;

        switch (command.type) {
          case 'move_task':
            coreResult = moveTaskWithCascade(command.taskId, new Date(command.startDate + 'T00:00:00.000Z'), this.snapshot, opts);
            break;
          case 'resize_task':
            coreResult = resizeTaskWithCascade(command.taskId, command.anchor, new Date(command.date + 'T00:00:00.000Z'), this.snapshot, opts);
            break;
          case 'recalculate_schedule':
            if (command.taskId) {
              coreResult = recalculateTaskFromDependencies(command.taskId, this.snapshot, opts);
            } else {
              coreResult = recalculateProjectSchedule(this.snapshot, opts);
            }
            break;
        }

        const result: ScheduleCommandResult = {
          changedTasks: coreResult.changedTasks,
          changedIds: coreResult.changedIds,
        };

        if (options?.includeSnapshot) {
          const map = new Map(this.snapshot.map(t => [t.id, t]));
          for (const changed of coreResult.changedTasks) {
            map.set(changed.id, changed);
          }
          result.snapshot = Array.from(map.values());
        }

        return result;
      }

      /** Legacy method — delegates to recalculateTaskFromDependencies for backward compat */
      recalculateDates(changedTaskId: string, skipStart?: boolean): Map<string, Task> {
        const opts: CoreOptions = {
          businessDays: this.options.businessDays,
          weekendPredicate: this.options.weekendPredicate,
        };
        const result = recalculateTaskFromDependencies(changedTaskId, this.snapshot, opts);
        return new Map(result.changedTasks.map(t => [t.id, t]));
      }
    }
    ```

    Key points:
    - No local date math, cascade, or hierarchy code — all delegated to gantt-lib
    - `snapshot` is `Task[]` (gantt-lib uses arrays, old scheduler used Map internally)
    - Type compatibility: gantt-lib's `Task` may have extra fields (`children`, `sortOrder`) — this is fine since MCP Task is a subset. Cast if needed for strict type mismatch on `locked` field.
    - The `recalculateDates` legacy method signature returns `Map<string, Task>` — callers in TaskService may use this.

    In `packages/mcp/src/services/task.service.ts`:
    - Verify the import of `TaskScheduler` still resolves (it will, since the class name and method signatures are unchanged)
    - No changes needed to TaskService unless the gantt-lib `Task` type has a `locked` field that MCP's doesn't — in that case, add `locked?: boolean` to MCP's Task interface in types.ts
