<task type="auto">
  <name>Create in-memory task store with CRUD operations</name>
  <files>src/store.ts</files>
  <action>
    Create src/store.ts with TaskStore class for in-memory task management:

    1. Import types:
       import { Task, CreateTaskInput, UpdateTaskInput } from './types.js';

    2. Create TaskStore class:
       export class TaskStore {
         private tasks: Map<string, Task>;

         constructor() {
           this.tasks = new Map();
         }

         // Create task with auto-generated UUID
         create(input: CreateTaskInput): Task {
           const id = crypto.randomUUID();
           const task: Task = {
             id,
             name: input.name,
             startDate: input.startDate,
             endDate: input.endDate,
             color: input.color,
             progress: input.progress ?? 0,
             dependencies: input.dependencies ?? [],
           };
           this.tasks.set(id, task);
           return task;
         }

         // List all tasks
         list(): Task[] {
           return Array.from(this.tasks.values());
         }

         // Get task by ID
         get(id: string): Task | undefined {
           return this.tasks.get(id);
         }

         // Update task
         update(id: string, input: UpdateTaskInput): Task | undefined {
           const task = this.tasks.get(id);
           if (!task) return undefined;

           const updated: Task = {
             ...task,
             ...(input.name !== undefined && { name: input.name }),
             ...(input.startDate !== undefined && { startDate: input.startDate }),
             ...(input.endDate !== undefined && { endDate: input.endDate }),
             ...(input.color !== undefined && { color: input.color }),
             ...(input.progress !== undefined && { progress: input.progress }),
             ...(input.dependencies !== undefined && { dependencies: input.dependencies }),
           };

           this.tasks.set(id, updated);
           return updated;
         }

         // Delete task
         delete(id: string): boolean {
           return this.tasks.delete(id);
         }
       }

    3. Export singleton instance:
       export const taskStore = new TaskStore();

    DO NOT add file persistence - in-memory only per STATE.md decision.
    DO use crypto.randomUUID() for ID generation (built-in Node module).
  </action>
