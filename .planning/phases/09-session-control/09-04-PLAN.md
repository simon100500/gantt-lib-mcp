```typescript
// packages/server/src/middleware/auth-middleware.ts

import type { FastifyRequest, FastifyReply } from 'fastify';

// Augment FastifyRequest so handlers can access user context via req.user
declare module 'fastify' {
  interface FastifyRequest {
    user?: {
      userId: string;
      email: string;
      projectId: string;
      sessionId: string;
    };
  }
}

/**
 * Fastify preHandler hook.
 * Extracts Bearer token from Authorization header.
 * Verifies JWT signature and expiry.
 * Attaches decoded payload to req.user.
 * Returns 401 if token missing, malformed, or expired.
 */
export async function authMiddleware(
  request: FastifyRequest,
  reply: FastifyReply
): Promise<void>
```

```typescript
  const tasks = await taskStore.list(req.user!.projectId);
  return reply.send(tasks);

  // ...
  runAgentWithHistory(message, req.user!.projectId, req.user!.sessionId).catch(...)
  // ...

fastify.delete('/api/tasks', { preHandler: [authMiddleware] }, async (req, reply) => {
  const count = await taskStore.deleteAll(req.user!.projectId);
  broadcastToSession(req.user!.sessionId, { type: 'tasks', tasks: [] });
  return reply.send({ deleted: count });
});
```
