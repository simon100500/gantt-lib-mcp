```typescript
// packages/web/src/hooks/useAuth.ts

export interface AuthUser {
  id: string;
  email: string;
}

export interface AuthProject {
  id: string;
  name: string;
}

export interface AuthState {
  isAuthenticated: boolean;
  user: AuthUser | null;
  project: AuthProject | null;
  accessToken: string | null;
  refreshToken: string | null;
  projects: AuthProject[];  // all projects for the user
}

export interface UseAuthResult extends AuthState {
  login(tokens: { accessToken: string, refreshToken: string }, user: AuthUser, project: AuthProject): void;
  logout(): Promise<void>;
  switchProject(projectId: string): Promise<void>;  // fetches new tokens for new project
  createProject(name: string): Promise<AuthProject>;
  refreshAccessToken(): Promise<string | null>;  // returns new access token or null on failure
}

export function useAuth(): UseAuthResult
```

- `switchProject(projectId)`: call POST /api/auth/switch-project (or re-verify? — simplest: POST /api/projects/:id/switch) — actually: fetch new tokens by calling a new server endpoint POST /api/auth/switch-project { projectId } with current Bearer token.
