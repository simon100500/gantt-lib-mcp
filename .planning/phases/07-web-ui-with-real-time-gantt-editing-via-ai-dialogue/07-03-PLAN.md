    2. Create packages/server/src/ws.ts:
       - Maintain a Set<WebSocket> of active connections
       - Export `broadcast(msg: ServerMessage): void` — sends JSON to all open connections
       - Export `registerWsRoutes(fastify: FastifyInstance): void` — registers GET /ws handler
       - On WS /ws connection: add socket to set, send `{type:'connected'}`, on close remove from set
       - On incoming message from client: parse JSON, if `type === 'chat'` emit an event for the agent runner
       - Use EventEmitter or simple callback registry to decouple WS from agent — export `onChatMessage(handler: (msg: string) => void)`

       ```typescript
       import { WebSocket } from 'ws';
       import type { FastifyInstance } from 'fastify';

       export type ServerMessage =
         | { type: 'token'; content: string }
         | { type: 'tasks'; tasks: unknown[] }
         | { type: 'error'; message: string }
         | { type: 'done' };

       const connections = new Set<WebSocket>();
       const chatHandlers: Array<(msg: string) => void> = [];

       export function broadcast(msg: ServerMessage): void {
         const json = JSON.stringify(msg);
         for (const socket of connections) {
           if (socket.readyState === WebSocket.OPEN) socket.send(json);
         }
       }

       export function onChatMessage(handler: (msg: string) => void): void {
         chatHandlers.push(handler);
       }

       export function registerWsRoutes(fastify: FastifyInstance): void {
         fastify.get('/ws', { websocket: true }, (socket) => {
           connections.add(socket);
           socket.send(JSON.stringify({ type: 'connected' }));
           socket.on('message', (raw) => {
             try {
               const data = JSON.parse(raw.toString()) as { type: string; message?: string };
               if (data.type === 'chat' && data.message) {
                 for (const h of chatHandlers) h(data.message);
               }
             } catch { /* ignore invalid JSON */ }
           });
           socket.on('close', () => connections.delete(socket));
         });
       }
       ```

       import websocket from '@fastify/websocket';

         } catch (err) {
           broadcast({ type: 'error', message: String(err) });
         }
       });

    ```typescript
    // packages/server/src/agent.ts
    import { query, isSDKResultMessage, isSDKAssistantMessage } from '@qwen-code/sdk';
    import { fileURLToPath } from 'url';
    import { dirname, join } from 'path';
    import { existsSync } from 'fs';
    import { readFile } from 'fs/promises';
    import * as dotenv from 'dotenv';
    import { taskStore } from '@gantt/mcp';
    import { broadcast } from './ws.js';

    const __filename = fileURLToPath(import.meta.url);
    const __dirname = dirname(__filename);
    // dist/agent.js → up 3 levels → project root
    const PROJECT_ROOT = join(__dirname, '../../..');
    dotenv.config({ path: join(PROJECT_ROOT, '.env') });

    function resolveEnv() {
      return {
        OPENAI_API_KEY:  process.env.OPENAI_API_KEY  ?? process.env.ANTHROPIC_AUTH_TOKEN  ?? '',
        OPENAI_BASE_URL: process.env.OPENAI_BASE_URL ?? 'https://api.z.ai/api/paas/v4/',
        OPENAI_MODEL:    process.env.OPENAI_MODEL    ?? process.env.ANTHROPIC_DEFAULT_SONNET_MODEL ?? 'glm-4.7',
      };
    }

    export async function runAgentWithHistory(userMessage: string): Promise<void> {
      // 1. Save user message to DB
      await taskStore.addMessage('user', userMessage);

      // 2. Load full conversation history from DB
      const messages = await taskStore.getMessages();

      // 3. Load system prompt
      const systemPromptPath = join(PROJECT_ROOT, 'packages/mcp/agent/prompts/system.md');
      const systemPrompt = existsSync(systemPromptPath)
        ? await readFile(systemPromptPath, 'utf-8')
        : 'You are a Gantt chart planning assistant.';

      // 4. Build prompt with history
      // Format: system prompt + past messages as context + current user message
      const historyContext = messages
        .slice(0, -1) // exclude the message we just added (it's the current turn)
        .map(m => `${m.role === 'user' ? 'User' : 'Assistant'}: ${m.content}`)
        .join('\n');

      const fullPrompt = [
        systemPrompt,
        historyContext.length > 0 ? `\n\n## Conversation history:\n${historyContext}` : '',
        `\n\nUser: ${userMessage}`,
      ].join('');

      const env = resolveEnv();
      const mcpServerPath = join(PROJECT_ROOT, 'packages/mcp/dist/index.js');
      const dbPath = process.env.DB_PATH ?? join(PROJECT_ROOT, 'gantt.db');

      // 5. Run agent session
      const session = query({
        prompt: fullPrompt,
        options: {
          model: env.OPENAI_MODEL,
          cwd: PROJECT_ROOT,
          env: {
            ...env,
            DB_PATH: dbPath,
          },
          mcpServers: [{
            type: 'stdio' as const,
            command: 'node',
            args: [mcpServerPath],
            env: { DB_PATH: dbPath },
          }],
        },
      });

      // 6. Stream tokens to WebSocket
      let assistantResponse = '';
      for await (const event of session) {
        if (isSDKAssistantMessage(event)) {
          const token = event.message.content ?? '';
          assistantResponse += token;
          broadcast({ type: 'token', content: token });
        }
        if (isSDKResultMessage(event)) {
          break;
        }
      }

      // 7. Save assistant response to DB
      if (assistantResponse) {
        await taskStore.addMessage('assistant', assistantResponse);
      }

      // 8. Broadcast updated tasks snapshot
      const tasks = await taskStore.list();
      broadcast({ type: 'tasks', tasks });

      // 9. Signal turn complete
      broadcast({ type: 'done' });
    }
    ```

    Key implementation notes:
    - The MCP server child process spawned by query() uses the same DB_PATH as the server,
      so task mutations by the AI are immediately visible via taskStore.list()
    - History is built from DB — model remembers context across browser refreshes/restarts
    - If OPENAI_API_KEY is not configured, throw an error and broadcast { type: 'error' }
    - Wrap entire function in try/catch, broadcast error on failure
