<!-- @libsql/client WASM API (no native binary — works on Windows without Build Tools) -->

```typescript
// Import pattern (ESM):
import { createClient } from '@libsql/client/web';
// OR for Node.js file:// URLs:
import { createClient } from '@libsql/client';

// Create a file-based SQLite client:
const client = createClient({ url: 'file:./gantt.db' });

// Execute SQL:
await client.execute('CREATE TABLE IF NOT EXISTS tasks (...)');

// Execute with params (positional):
await client.execute({
  sql: 'INSERT INTO tasks (id, name, start_date, end_date) VALUES (?, ?, ?, ?)',
  args: [id, name, startDate, endDate]
});

// Batch:
await client.batch([
  { sql: 'DELETE FROM tasks WHERE id = ?', args: [id] },
  { sql: 'DELETE FROM dependencies WHERE task_id = ?', args: [id] }
], 'write');

// Query:
const result = await client.execute('SELECT * FROM tasks');
// result.rows: Row[] — each row is array-like with named properties
// result.rows[0].id, result.rows[0].name, etc.
```

    DB path resolved from env var DB_PATH, defaulting to './gantt.db' relative to CWD.

    ```typescript
    // packages/mcp/src/db.ts
    import { createClient, type Client } from '@libsql/client';

    let _db: Client | null = null;

    export type DbClient = Client;

    export async function getDb(): Promise<Client> {
      if (_db) return _db;

      const dbPath = process.env.DB_PATH ?? './gantt.db';
      _db = createClient({ url: `file:${dbPath}` });

      await _db.execute(`
        CREATE TABLE IF NOT EXISTS tasks (
          id TEXT PRIMARY KEY,
          name TEXT NOT NULL,
          start_date TEXT NOT NULL,
          end_date TEXT NOT NULL,
          color TEXT,
          progress REAL DEFAULT 0
        )
      `);

      await _db.execute(`
        CREATE TABLE IF NOT EXISTS dependencies (
          id TEXT PRIMARY KEY,
          task_id TEXT NOT NULL REFERENCES tasks(id) ON DELETE CASCADE,
          dep_task_id TEXT NOT NULL,
          type TEXT NOT NULL CHECK(type IN ('FS','SS','FF','SF')),
          lag REAL DEFAULT 0
        )
      `);

      await _db.execute(`
        CREATE TABLE IF NOT EXISTS messages (
          id TEXT PRIMARY KEY,
          role TEXT NOT NULL CHECK(role IN ('user','assistant')),
          content TEXT NOT NULL,
          created_at TEXT NOT NULL
        )
      `);

      return _db;
    }
    ```

    Key design notes:
    - Singleton pattern — one client per process lifetime
    - dependencies.task_id has ON DELETE CASCADE so deleting a task removes its deps
    - dependencies.dep_task_id is a reference (no FK enforcement for flexibility — SQLite needs PRAGMA foreign_keys=ON which complicates batches)
    - progress stored as REAL (0.0-100.0)
    - messages.created_at stored as ISO timestamp TEXT

       // After insert/update, reload all tasks from DB into memory, run scheduler, write back
       private async runScheduler(changedTaskId: string, skipStart = false): Promise<void> {
         const tasks = await this.list();
         const taskMap = new Map(tasks.map(t => [t.id, t]));
         const updates = this.scheduler.recalculateDates(changedTaskId, skipStart, taskMap);
         for (const [id, updated] of updates) {
           await this.writeTaskToDB(id, updated);
         }
       }
