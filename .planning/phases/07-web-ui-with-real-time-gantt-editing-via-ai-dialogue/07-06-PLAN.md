<!-- Multi-stage Dockerfile pattern for Node + Nginx -->

```dockerfile
# Stage 1: Build web
FROM node:22-alpine AS build-web
WORKDIR /app
COPY package*.json ./
COPY packages/web/package.json ./packages/web/
# ... install and build

# Stage 2: Build server
FROM node:22-alpine AS build-server
# ... install and build

# Stage 3: Runtime (Nginx + Node)
FROM nginx:alpine AS runtime
RUN apk add --no-cache nodejs npm supervisor
# Copy static files
COPY --from=build-web /app/packages/web/dist /usr/share/nginx/html
# Copy server
COPY --from=build-server /app/packages/server/dist /app/server/dist
# Nginx config
COPY nginx.conf /etc/nginx/conf.d/default.conf
# Start both nginx and node via supervisord or shell script
```

    1. Create Dockerfile (multi-stage build):

       ```dockerfile
       # ── Stage 1: Build web (React + Vite) ──────────────────────────────────────
       FROM node:22-alpine AS build-web
       WORKDIR /build

       # Copy only package manifests first for layer caching
       COPY package.json package-lock.json ./
       COPY packages/web/package.json ./packages/web/

       # Install all workspace deps (needed for dhtmlx-gantt, vite, etc.)
       RUN npm ci --ignore-scripts

       # Copy web source
       COPY packages/web ./packages/web

       # Build web
       RUN npm run build -w packages/web


       # ── Stage 2: Build server + mcp ────────────────────────────────────────────
       FROM node:22-alpine AS build-server
       WORKDIR /build

       COPY package.json package-lock.json ./
       COPY packages/mcp/package.json ./packages/mcp/
       COPY packages/server/package.json ./packages/server/

       RUN npm ci --ignore-scripts

       COPY packages/mcp ./packages/mcp
       COPY packages/server ./packages/server

       # Build mcp first (server depends on it)
       RUN npm run build:mcp
       RUN npm run build:server


       # ── Stage 3: Runtime ───────────────────────────────────────────────────────
       FROM nginx:1.27-alpine AS runtime

       # Add Node.js runtime (for Fastify server)
       RUN apk add --no-cache nodejs npm

       WORKDIR /app

       # Copy static web build
       COPY --from=build-web /build/packages/web/dist /usr/share/nginx/html

       # Copy compiled server
       COPY --from=build-server /build/packages/server/dist ./server/dist
       COPY --from=build-server /build/packages/mcp/dist ./mcp/dist

       # Copy production node_modules (server deps only)
       COPY --from=build-server /build/node_modules ./node_modules
       COPY --from=build-server /build/packages/server/node_modules ./packages/server/node_modules
       COPY --from=build-server /build/packages/mcp/node_modules ./packages/mcp/node_modules

       # Copy MCP agent prompts (needed by agent runner)
       COPY --from=build-server /build/packages/mcp/agent/prompts ./mcp/agent/prompts

       # Nginx config
       COPY nginx.conf /etc/nginx/conf.d/default.conf

       # Startup script
       COPY docker-entrypoint.sh /docker-entrypoint.sh
       RUN chmod +x /docker-entrypoint.sh

       # Persistent data directory (mount CapRover Persistent Directory here)
       VOLUME /data

       EXPOSE 80

       CMD ["/docker-entrypoint.sh"]
       ```

       NOTE: @libsql/client WASM does not require native compilation, so apk add python3/make/g++ is NOT needed. This is a key advantage of using @libsql/client over better-sqlite3.

    2. Create docker-entrypoint.sh (starts both Nginx and Node):
       ```bash
       #!/bin/sh
       set -e

       # Start Fastify server in background
       DB_PATH="${DB_PATH:-/data/gantt.db}" \
       PORT=3000 \
       node /app/server/dist/index.js &

       # Start Nginx in foreground
       nginx -g "daemon off;"
       ```

    3. Create nginx.conf:
       ```nginx
       server {
           listen 80;
           server_name _;

           # React SPA — serve index.html for all non-API routes
           location / {
               root /usr/share/nginx/html;
               try_files $uri $uri/ /index.html;
           }

           # WebSocket proxy (must be before /api to match /ws)
           location /ws {
               proxy_pass http://localhost:3000;
               proxy_http_version 1.1;
               proxy_set_header Upgrade $http_upgrade;
               proxy_set_header Connection "Upgrade";
               proxy_set_header Host $host;
               proxy_read_timeout 86400s;
           }

           # REST API proxy
           location /api/ {
               proxy_pass http://localhost:3000;
               proxy_set_header Host $host;
               proxy_set_header X-Real-IP $remote_addr;
               proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
           }
       }
       ```

    4. Create .env.example:
       ```
       # Required: AI API key (Z.AI or OpenAI-compatible)
       OPENAI_API_KEY=your-api-key-here

       # Optional: AI model (default: glm-4.7 via Z.AI)
       OPENAI_BASE_URL=https://api.z.ai/api/paas/v4/
       OPENAI_MODEL=glm-4.7

       # Optional: SQLite DB path (default: ./gantt.db, in container: /data/gantt.db)
       DB_PATH=/data/gantt.db

       # Optional: Server port (default: 3000, in container Nginx handles :80)
       PORT=3000
       ```
