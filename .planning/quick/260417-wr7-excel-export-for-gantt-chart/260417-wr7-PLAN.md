```typescript
import * as XLSX from 'xlsx';
import type { Task } from '../types';

export type ExcelDetailLevel = 'basic' | 'full';

export interface ExcelExportOptions {
  fileName?: string;
  detailLevel?: ExcelDetailLevel;
  projectName?: string;
}

function flattenTasks(tasks: Task[]): Task[] {
  // Return tasks in sortOrder order; if no sortOrder, preserve array order
  return [...tasks].sort((a, b) => (a.sortOrder ?? 0) - (b.sortOrder ?? 0));
}

function formatDate(date: string | Date): string {
  if (date instanceof Date) {
    return date.toISOString().slice(0, 10);
  }
  return String(date).slice(0, 10);
}

export function exportToExcel(tasks: Task[], options: ExcelExportOptions = {}): void {
  const { fileName = 'export.xlsx', detailLevel = 'basic', projectName } = options;
  const flat = flattenTasks(tasks);

  if (detailLevel === 'basic') {
    // Columns: #, Name, Start Date, End Date, Type
    const rows = flat.map((t, i) => ({
      '#': i + 1,
      'Название': t.name,
      'Начало': formatDate(t.startDate),
      'Окончание': formatDate(t.endDate),
      'Тип': t.type === 'milestone' ? 'Веха' : 'Задача',
    }));
    const ws = XLSX.utils.json_to_sheet(rows);
    ws['!cols'] = [{ wch: 5 }, { wch: 40 }, { wch: 12 }, { wch: 12 }, { wch: 8 }];
    const wb = XLSX.utils.book_new();
    XLSX.utils.book_append_sheet(wb, ws, projectName || 'Задачи');
    XLSX.writeFile(wb, fileName);
  } else {
    // Full: +Progress, Dependencies, Parent
    const taskMap = new Map(flat.map((t) => [t.id, t]));
    const rows = flat.map((t, i) => ({
      '#': i + 1,
      'Название': t.name,
      'Начало': formatDate(t.startDate),
      'Окончание': formatDate(t.endDate),
      'Тип': t.type === 'milestone' ? 'Веха' : 'Задача',
      'Прогресс': t.progress != null ? `${Math.round(t.progress)}%` : '',
      'Зависимости': t.dependencies?.map((d) => {
        const depTask = taskMap.get(d.taskId);
        const depName = depTask?.name ?? d.taskId;
        return `${d.type}${d.lag ? `+${d.lag}д` : ''} ${depName}`;
      }).join('; ') || '',
      'Родитель': t.parentId ? (taskMap.get(t.parentId)?.name ?? t.parentId) : '',
    }));
    const ws = XLSX.utils.json_to_sheet(rows);
    ws['!cols'] = [{ wch: 5 }, { wch: 40 }, { wch: 12 }, { wch: 12 }, { wch: 8 }, { wch: 8 }, { wch: 30 }, { wch: 30 }];
    const wb = XLSX.utils.book_new();
    XLSX.utils.book_append_sheet(wb, ws, projectName || 'Задачи');
    XLSX.writeFile(wb, fileName);
  }
}
```

Key points:
- No server round-trip. SheetJS runs entirely client-side.
- Russian column headers matching existing UI language.
- `flattenTasks` respects sortOrder for row ordering.
- Basic = 5 columns (id, name, dates, type). Full = 8 columns (+progress, deps, parent).
- Dependencies formatted as "FS+2д Task Name" for readability.
